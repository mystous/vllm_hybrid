%% Fig.6 — 24 후보 기법 5 분류군 dendrogram

\begin{figure}[t]
\centering
\begin{tikzpicture}[
    x=0.55cm, y=0.55cm,
    root/.style={rectangle, draw, thick, fill=blue!20, font=\small, align=center,
                 minimum width=3.5cm, minimum height=0.7cm},
    class/.style={rectangle, draw, fill=yellow!15, font=\footnotesize, align=center,
                  minimum width=2.4cm, minimum height=0.6cm},
    leaf/.style={font=\scriptsize, align=center},
    pos/.style={text=green!50!black},
    neg/.style={text=red!60!black},
]
% Root
\node[root] (root) at (8, 7) {24 CPU optimization\\candidates};

% 5 classes
\node[class, fill=red!10] (A) at (1, 4.5) {Class A\\Direct kernel\\(7 cand.)};
\node[class, fill=red!10] (B) at (5, 4.5) {Class B\\System isolation\\(4 cand.)};
\node[class, fill=red!10] (C) at (9, 4.5) {Class C\\Sub-step sched.\\(3 cand.)};
\node[class, fill=red!10] (D) at (13, 4.5) {Class D\\Aux. workload\\(4 cand.)};
\node[class, fill=green!12] (E) at (16.5, 4.5) {Class E\\Concurrent aux.\\(5 cand. + 2$\times$2)};

% Lines from root to classes
\foreach \child in {A, B, C, D, E} {
    \draw[thick] (root.south) -- (\child.north);
}

% Leaves under each class — sample representative
\node[leaf, neg] (a1) at (1, 3.0) {AVX-512 tokenizer};
\node[leaf, neg] (a2) at (1, 2.4) {AMX matmul};
\node[leaf, neg] (a3) at (1, 1.8) {Jacobi router (--94\%)};

\node[leaf, pos] (b1) at (5, 3.0) {NUMA pin};
\node[leaf, neg] (b2) at (5, 2.4) {cgroup+hugepages};
\node[leaf, pos] (b3) at (5, 1.8) {NUMA $\oplus$ aux};

\node[leaf, neg] (c1) at (9, 3.0) {task-pool stub};
\node[leaf, neg] (c2) at (9, 2.4) {task-pool fill (--14\%)};

\node[leaf, neg] (d1) at (13, 3.0) {AMX draft proxy};
\node[leaf, neg] (d2) at (13, 2.4) {cold-KV long ctx};

\node[leaf, pos] (e1) at (16.5, 3.0) {\textbf{branchy 100ms}};
\node[leaf] (e2) at (16.5, 2.4) {regular 100ms};
\node[leaf, neg] (e3) at (16.5, 1.8) {tokenize 20ms};

% Class-to-leaf lines
\foreach \c/\l in {A/a1, A/a2, A/a3, B/b1, B/b2, B/b3, C/c1, C/c2, D/d1, D/d2, E/e1, E/e2, E/e3} {
    \draw[gray!50] (\c.south) -- (\l.north);
}

% Legend
\node[leaf, pos] at (8, 0.4) {green = positive ($\Delta > 0$)};
\node[leaf, neg] at (12, 0.4) {red = negative ($\Delta \leq 0$)};

\end{tikzpicture}
\caption{24 후보 최적화 기법의 5 분류군 dendrogram. 분류군 A/C/D는 전부 net-negative, 분류군 B는 부분 양성 (NUMA pin만 robust), 분류군 E는 유일하게 paper-main 임계점 도달 (branchy $\times$ 100\,ms).}
\label{fig:class-taxonomy}
\end{figure}
