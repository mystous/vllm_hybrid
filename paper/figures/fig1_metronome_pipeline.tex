\begin{figure}[t]
\centering
\resizebox{0.92\columnwidth}{!}{%
\begin{tikzpicture}[
    node distance=0.35cm,
    stage/.style={rectangle, draw, rounded corners=2pt,
                  minimum width=4.4cm, minimum height=0.62cm,
                  align=center, font=\small, fill=blue!10},
    serve/.style={rectangle, draw, rounded corners=2pt,
                  minimum width=4.4cm, minimum height=0.62cm,
                  align=center, font=\small, fill=green!12},
    decision/.style={diamond, draw, aspect=2.4,

\node[stage, below=of s3]     (s4) {\textsc{Accent} : pattern by $\rho_{\text{pcie}}$};
\node[serve, below=of s4]     (s5) {\textsc{Ritardando} : adaptive serve};

    node[left=2pt, font=\scriptsize, pos=0.5] {no} (s5.west);
\end{tikzpicture}%
}
