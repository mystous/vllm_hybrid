%% figures/fig2_step_alignment.tex
%% Fig.2 — GPU step boundary와 CPU 보조 작업의 시간 정렬 timing diagram (TikZ)

\begin{figure}[t]
\centering
\begin{tikzpicture}[
    x=0.42cm, y=0.65cm,
    gpu/.style={rectangle, draw, fill=blue!18, minimum height=0.55cm, font=\footnotesize},
    cpu/.style={rectangle, draw, fill=green!18, minimum height=0.40cm, font=\footnotesize},
    idle/.style={rectangle, draw, dashed, fill=gray!8, minimum height=0.55cm, font=\footnotesize},
    label/.style={font=\footnotesize, align=center},
    axis/.style={-{Stealth[length=1.8mm]}, thick},
]
% Time axis
\draw[axis] (-0.3,0) -- (22.5, 0) node[right, font=\footnotesize] {time (ms)};
\foreach \x/\t in {0/0, 4/40, 8/80, 12/120, 16/160, 20/200} {
    \draw (\x, 0) -- (\x, -0.10);
    \node[below, font=\scriptsize] at (\x, -0.10) {\t};
}

% GPU steps (40 ms each) — 5 steps
\foreach \i in {0,1,2,3,4} {
    \pgfmathsetmacro{\xs}{\i*4}
    \pgfmathsetmacro{\xe}{(\i+1)*4}
    \node[gpu, anchor=south west, minimum width={(\xe-\xs)*0.42cm-1pt}, font=\scriptsize]
        at (\xs, 0.18) {GPU step};
}
\node[label, anchor=east] at (-0.4, 0.45) {GPU};

% CPU auxiliary work — period 100ms (k=2.5)
% Active windows: ms 0-30, ms 100-130 (just illustrative)
\node[cpu, anchor=south west, minimum width={3*0.42cm-1pt}] at (0, 1.2) {CPU aux};
\node[cpu, anchor=south west, minimum width={3*0.42cm-1pt}] at (10, 1.2) {CPU aux};
\node[cpu, anchor=south west, minimum width={3*0.42cm-1pt}] at (20, 1.2) {(next)};
\node[label, anchor=east] at (-0.4, 1.45) {CPU};

% Cycle marker
\draw[<->, thick, gray!70!black] (0, 2.1) -- (10, 2.1)
    node[midway, above, font=\footnotesize, text=gray!50!black] {$\tcpu = k \cdot \tgpu = 100$\,ms};
\draw[<->, thick, gray!70!black] (0, -0.7) -- (4, -0.7)
    node[midway, below, font=\footnotesize, text=gray!50!black] {$\tgpu \approx 40$\,ms};

% Step boundary alignment markers
\foreach \x in {0, 4, 8, 12, 16, 20} {
    \draw[dotted, gray!50] (\x, -0.7) -- (\x, 2.1);
}

\end{tikzpicture}
\caption{\textsc{Meter} 단계의 주기 정렬. GPU step 주기 $\tgpu \approx 40$\,ms에 대해 CPU 보조 작업 주기 $\tcpu$를 정수배 $k=2.5$ (즉 $\tcpu = 100$\,ms)로 정렬한다. 보조 작업은 step boundary에 정확히 정렬되어 critical path를 침범하지 않으며, idle window의 amortize 효율을 최대화한다.}
\label{fig:step-alignment}
\end{figure}
